### 使用 set

```javascript
function unique(array) {
    return [...new Set(array)];
}
```

### 使用 Map

```javascript
function unique(array) {
    const seen = new Map();
    return array.filter(a => !seen.has(a) && seen.set(a, 1));
}
```

### 双层循环

```javascript
function unique(arr) {
    var len = arr.length;
    var _arr = [];
    var _len;
    for (let i = 0; i < len; i++) {
        for (let j = 0, _len = _arr.length; j < _len; j++) {
            if (arr[i] === _arr[j]) {
                break;
            }
        }
        if (j === _len) {
            _arr.push(arr[i]);
        }
    }
    return _arr;
}
```

### 使用 indexOf 简化内层循环

```javascript
function unique(arr) {
    var len = arr.length;
    var _arr = [];
    for (let i = 0; i < len; i++) {
        if (_arr.indexOf(arr[i]) === -1) {
            _arr.push(arr[i]);
        }
    }
}
```

### 纯数值数组，排序后去重

```javascript
function unique(arr) {
    //归并排序
    var mergeSort = {
        merge: function(left, right) {
            var tmp = [];
            while (left.length && right.length) {
                if (left[0] < right[0]) tmp.push(left.shift());
                else tmp.push(right.shift());
            }
            return tmp.concat(left, right);
        },
        sort: function(array) {
            if (array.length === 1) return array;
            var mid = Math.floor(array.length / 2);
            var left = array.slice(0, mid);
            var right = array.slice(mid);
            return this.merge(this.sort(left), this.sort(right));
        }
    };

    var arrSort = mergeSort.sort(arr);
    var _arr = [];
    var seen;
    for (var i = 0; i < arrSort.length; i++) {
        if (!i || seen !== arrSort[i]) {
            _arr.push(arrSort[i]);
        }
        seen = _arr[_arr.length - 1];
    }
    return _arr;
}
```
